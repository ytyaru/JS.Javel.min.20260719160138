``````markdown
## 1. トークナイズ設計の論理的アプローチ## 単一責任の原則に基づくコンポーネント分割
1字ずつ読み取るパーサコンビネータ方式で「フェンス」と「ブロック」を分離するため、責務を極限まで細分化します。

   1. Cursor（入力管理の責任）: 原稿の現在位置（インデックス）を保持し、1文字進める・先読み（Peek）する機能を提供します。
   2. Parser（最小単位の判定責任）: 特定の文字パターン（例：3つ以上の連続した特定の文字など）にマッチするかどうかだけを判定する関数（コンビネータ）の集合です。
   3. Tokenizer（トークン生成・判定順序の制御責任）: 各Parserを組み合わせ、入力文字列から適切な Token を切り出して生成します。

## フェンスとブロックの分離論理

* フェンス（FenceToken）: 特定の囲み文字（例：--- や ```）で始まり、終了条件（同じ囲み文字、または原稿の終端）を持つ構造です。行頭から始まる明確な境界を持ちます。
* ブロック（BlockToken）: フェンス以外の行単位、または一連のテキストの塊です。

これらを分離するため、トークナイズの基本ループでは「まずフェンスの開始条件を満たすか」を走査し、満たさない場合は「次のフェンス開始、またはブロックの区切り（改行など）まで」をブロックとして処理する論理構造を採用します。
------------------------------
## 2. 再利用可能部品のコード実装## token.js （前回の型定義の維持）
main.js からインポートできるように、前回のインターフェース定義を前提とします。
## tokenizer.ts

import { BaseToken, FenceToken, BlockToken, LinerRange } from './token.js';
/**
 * 原稿の読み込み位置を管理するクラス（入力管理の単一責任）
 */class Cursor {
    private index: number = 0;
    constructor(private readonly text: string) {}

    get position(): number { return this.index; }
    get isEof(): boolean { return this.index >= this.text.length; }

    peek(offset: number = 0): string {
        return this.text.charAt(this.index + offset);
    }

    advance(count: number = 1): void {
        this.index += count;
    }

    substring(start: number, end: number): string {
        return this.text.substring(start, end);
    }
}
/**
 * パーサコンビネータの基底ユーティリティ（パターンマッチの責任）
 */class Combinators {
    /**
     * 特定の文字が指定回数以上連続しているかを判定し、その個数を返す
     */
    static matchRepeatedChar(cursor: Cursor, char: string, minCount: number): number {
        let count = 0;
        while (cursor.peek(count) === char) {
            count++;
        }
        return count >= minCount ? count : 0;
    }

    /**
     * 改行（LF）または終端までの範囲を走査する
     */
    static scanUntilLfOrEof(cursor: Cursor): void {
        while (!cursor.isEof && cursor.peek() !== '\n') {
            cursor.advance();
        }
    }
}
/**
 * トークン生成の制御クラス（トークナイズの責任）
 */export class Tokenizer {
    constructor(options?: any) {
        // オプションによる動作変更が必要な場合はここで保持します
    }

    /**
     * 文字列からトークンを順次生成するジェネレーター
     */
    *tokenize(manuscript: string): Generator<BaseToken, void, unknown> {
        const cursor = new Cursor(manuscript);

        while (!cursor.isEof) {
            // 1. フェンスの解析を試みる
            const fenceToken = this.tryParseFence(cursor);
            if (fenceToken) {
                yield fenceToken;
                continue;
            }

            // 2. フェンスでない場合はブロックとして処理する
            const blockToken = this.parseBlock(cursor);
            if (blockToken) {
                yield blockToken;
            }
        }
    }

    /**
     * フェンス構造の解析を試みる内部メソッド
     */
    private tryParseFence(cursor: Cursor): FenceToken | null {
        const startPos = cursor.position;
        const char = cursor.peek();

        // フェンスの代表例として「-」または「`」の3文字以上の連続を検証対象とする
        if (char !== '-' && char !== '`') {
            return null;
        }

        const num = Combinators.matchRepeatedChar(cursor, char, 3);
        if (num === 0) {
            return null;
        }

        // 開始フェンスの確定
        const prefixStart = startPos;
        cursor.advance(num);
        Combinators.scanUntilLfOrEof(cursor);
        if (cursor.peek() === '\n') { cursor.advance(); } // LF消費
        const prefixEnd = cursor.position;

        // コンテンツの走査（同じフェンスが出現するか、EOFまで）
        const contentStart = cursor.position;
        let suffixStart = cursor.position;
        let suffixEnd = cursor.position;
        let hasEndFence = false;

        while (!cursor.isEof) {
            const currentLineStart = cursor.position;
            const currentNum = Combinators.matchRepeatedChar(cursor, char, num);

            if (currentNum >= num) {
                suffixStart = currentLineStart;
                cursor.advance(currentNum);
                Combinators.scanUntilLfOrEof(cursor);
                if (cursor.peek() === '\n') { cursor.advance(); }
                suffixEnd = cursor.position;
                hasEndFence = true;
                break;
            }

            // 行を進める
            Combinators.scanUntilLfOrEof(cursor);
            if (cursor.peek() === '\n') { cursor.advance(); }
        }

        const contentEnd = hasEndFence ? suffixStart : cursor.position;

        return {
            kind: 'fence',
            range: { start: startPos, end: cursor.position },
            enclosure: { char, num },
            prefix: { range: { start: prefixStart, end: prefixEnd } },
            content: { range: { start: contentStart, end: contentEnd } },
            suffix: { range: { start: suffixStart, end: suffixEnd } }
        };
    }

    /**
     * ブロック（次の構造の開始、または改行・終端まで）を解析する内部メソッド
     */
    private parseBlock(cursor: Cursor): BlockToken | null {
        const startPos = cursor.position;

        // 次のフェンスの開始が来るか、原稿の終端に達するまで文字を進める
        while (!cursor.isEof) {
            const nextChar = cursor.peek();
            
            // 次の行頭などでフェンスが始まる可能性があるか検知
            if (nextChar === '-' || nextChar === '`') {
                const num = Combinators.matchRepeatedChar(cursor, nextChar, 3);
                if (num >= 3) {
                    // ここでブロックを終了し、次のフェンス処理へ譲る
                    break;
                }
            }

            // 通常の文字として消費
            cursor.advance();
        }

        const endPos = cursor.position;
        if (startPos === endPos) {
            return null;
        }

        return {
            kind: 'block',
            range: { start: startPos, end: endPos }
        };
    }
}

## 3. 仕様の根拠## UTF-8 と NFC に関する仕様
JavaScriptおよびTypeScriptの String 型は、内部的にUTF-16コードユニットのシーケンスとして文字列を扱います。

* 原稿が UTF-8（NFCに正規化済み） である場合、サロゲートペア（絵文字や一部の稀少漢字など）を除き、多くの文字は1つのインデックス（文字）として charAt や peek で正しく1文字ずつ読み取ることが可能です。
* サロゲートペアを厳密に考慮する場合は、cursor.peek() において String.prototype.codePointAt() や Array.from() などの仕組みを用いてコードポイント単位で進める必要がありますが、フェンスの記号（- や `` `）はASCII領域（U+002D, U+0060）であるため、通常の文字インデックス走査で干渉せずに判定できます。
* [MDN Web Docs: String.prototype.charAt()](https://developer.mozilla.org/ja/docs/Web/JavaScript/Reference/Global_Objects/String/charAt) において、charAt() はUTF-16コードユニットを返す仕様が記述されています。

## LF（改行コード）に関する仕様

* 原稿がLF（\n）に統一されているため、CRLF（\r\n）の考慮を省き、cursor.peek() === '\n' の判定のみで正確に行末（改行）を特定できる論理が成立します。

原稿は人間が読みやすい文書構造になったテキストである。なのでフェンス、ブロック間は必ず二連続以上改行で分離されている。もし二連続より大きな改行が続いていたら`SpaceToken`にすべきだ。その際の`count`は`行数-2`である。3連続改行であれば`3-2=1`だ。

# 解答

``````markdown
追加の仕様、および将来の拡張性に関する重要な前提を共有いただきありがとうございます。
ご提示いただいた「複数行にわたる接頭辞・接尾辞」および「フェンスのネスト」を含む仕様は、一般的なMarkdown等よりもはるかに高度な文脈解析を必要とします。
ご質問の「解析の実現可能性」に対する論理的な回答と、単一責任の原則・DRY原則を極限まで追求した「トークナイザのクラス分割設計」を以下に提示します。
------------------------------
## 1. 複数行の接辞を持つフェンス解析は困難か？
結論から申し上げると、通常の「1字ずつ進む前向き走査（LL(1)等のパーサ）」だけでは極めて困難ですが、「2段階（2パス）の走査」または「バックトラック（巻き戻し）」を導入することで論理的に破綻なく実装可能です。
## 困難とされる理由（論理的説明）
最悪のケースとして、以下のような構造が入力された場合を考えます。

＝＝＝＝＝
A（これが接頭辞なのか、単なるブロックテキストなのか？）
＝＝＝＝＝
B（これがフェンス内容なのか、それとも接尾辞なのか？）
＝＝＝＝＝

1遍の走査（シングルパス）で上から順に処理しようとすると、2つ目の ＝＝＝＝＝ に到達した時点で、それが「接頭辞の終わり」なのか「フェンス全体の終わり」なのかを確定（極定）できません。

## 解決のための論理的アプローチ
この問題を解決するためには、「まず外枠の境界（最も外側のフェンス記号）をすべて特定し、その後に内部を切り分ける」 というアプローチを取ります。

   1. フェンス記号（＝＝＝等）の最外殻のペアを探索する:
   行頭にあるフェンス記号を走査し、同じ長さ、あるいはそれ以上の長さの「閉じフェンス記号」がどこにあるかを、ドキュメント全体または後続行から先に見つけ出します。
   2. 境界の確定後に中身をパースする:
   「ここからここまでが1つの巨大なフェンス構造である」と外枠が確定すれば、その内部にある ＝＝＝＝ などの記号は「接頭辞や接尾辞の区切り」として一意に認識できるようになります。

したがって、責任を適切に分離すれば、将来的にこの拡張が入ってもパーサ全体の基本設計を壊さずに拡張可能です。
------------------------------

## 2. 単一責任・DRYを追求したトークナイザの分割設計
ご提案の通り、Fence、Block、Space をそれぞれ独立したクラス（子トークナイザ）へと完全に分離します。これらは共通のインターフェースを実装し、メインの Tokenizer がそれらを統括する設計にします。

## 共通の型定義・インターフェース (tokenizer-interface.ts)
すべてのTokenizerが満たすべき共通の契約（インターフェース）を定義します。

```typescript
import { BaseToken } from './token.js';

/**
 * 共通のカーソルインターフェース
 */
export interface TokenizerCursor {
    readonly text: string;
    readonly position: number;
    readonly isEof: boolean;
    peek(offset?: number): string;
    advance(count?: number): void;
    substring(start: number, end: number): string;
}

/**
 * 子トークナイザが実装すべき共通のインターフェース
 */
export interface SubTokenizer {
    /**
     * 現在のカーソル位置から自身のトークンとして解析可能か判定する
     */
    match(cursor: TokenizerCursor): boolean;
    /**
     * トークンを解析して生成する
     */
    parse(cursor: TokenizerCursor): BaseToken;
}

## ② 2連続以上の改行（スペース）を処理する責任: SpaceTokenizer
「3連続改行であれば 3 - 2 = 1」の仕様を厳密に計算します。また、フェンスやブロックの境界として機能する「2連続改行」自体は、このトークナイザが消費して区切りとします。

```typescript
import { SubTokenizer, TokenizerCursor } from './tokenizer-interface.js';
import { SpaceToken } from './token.js';

export class SpaceTokenizer implements SubTokenizer {
    match(cursor: TokenizerCursor): boolean {
        // 2連続以上の改行（LF）から始まる場合のみマッチ
        return cursor.peek(0) === '\n' && cursor.peek(1) === '\n';
    }

    parse(cursor: TokenizerCursor): SpaceToken {
        const startPos = cursor.position;
        let lfCount = 0;

        // 連続する改行をすべてカウントして消費する
        while (!cursor.isEof && cursor.peek() === '\n') {
            lfCount++;
            cursor.advance(1);
        }

        const endPos = cursor.position;
        // 仕様: count = 行数（LFの数） - 2
        const tokenCount = lfCount - 2;

        return {
            kind: 'space',
            range: { start: startPos, end: endPos },
            count: tokenCount
        };
    }
}
```

## ③ フェンス構造を処理する責任: FenceTokenizer
現時点では将来の複雑な接辞解析を「保留」とし、最も単純な「同じ長さのフェンス記号で挟まれた構造」のみを単一責任として実装します。将来拡張する際はこのクラスの内部ロジック、あるいは具象クラスだけを差し替えます。


```typescript
import { SubTokenizer, TokenizerCursor } from './tokenizer-interface.js';
import { FenceToken } from './token.js';

export class FenceTokenizer implements SubTokenizer {
    match(cursor: TokenizerCursor): boolean {
        // フェンスは必ず行頭始まり（今回は簡易的に3連続以上の特定の記号を想定。例として「＝」や「-」）
        const char = cursor.peek(0);
        if (char !== '＝' && char !== '-' && char !== '`') {
            return false;
        }
        return cursor.peek(1) === char && cursor.peek(2) === char;
    }

    parse(cursor: TokenizerCursor): FenceToken {
        const startPos = cursor.position;
        const char = cursor.peek();
        
        // 連続するフェンス記号の数を数える
        let num = 0;
        while (cursor.peek(num) === char) {
            num++;
        }

        // 開始フェンス行の終了（LF）まで進める
        const prefixStart = startPos;
        cursor.advance(num);
        while (!cursor.isEof && cursor.peek() !== '\n') { cursor.advance(1); }
        if (cursor.peek() === '\n') { cursor.advance(1); }
        const prefixEnd = cursor.position;

        const contentStart = cursor.position;
        let contentEnd = cursor.position;
        let suffixStart = cursor.position;
        let suffixEnd = cursor.position;

        // 閉じフェンスの探索
        while (!cursor.isEof) {
            const currentPos = cursor.position;
            
            // 行頭で同じ文字が同じ数以上連続しているか
            let matchCount = 0;
            while (cursor.peek(matchCount) === char) {
                matchCount++;
            }

            if (matchCount >= num) {
                contentEnd = currentPos;
                suffixStart = currentPos;
                cursor.advance(matchCount);
                while (!cursor.isEof && cursor.peek() !== '\n') { cursor.advance(1); }
                if (cursor.peek() === '\n') { cursor.advance(1); }
                suffixEnd = cursor.position;
                break;
            }

            // 次の行へ
            while (!cursor.isEof && cursor.peek() !== '\n') { cursor.advance(1); }
            if (cursor.peek() === '\n') { cursor.advance(1); }
            contentEnd = cursor.position;
        }

        return {
            kind: 'fence',
            range: { start: startPos, end: cursor.position },
            enclosure: { char, num },
            prefix: { range: { start: prefixStart, end: prefixEnd } },
            content: { range: { start: contentStart, end: contentEnd } },
            suffix: { range: { start: suffixStart, end: suffixEnd } }
        };
    }
}
```

## ④ ブロックを処理する責任: BlockTokenizer
「フェンスでもなく、2連続改行（スペース）でもない」テキストの塊を処理します。次のフェンスの開始、または2連続改行が出現した時点で処理の責任を終えます。

```typescript
import { SubTokenizer, TokenizerCursor } from './tokenizer-interface.js';
import { BlockToken } from './token.js';

export class BlockTokenizer implements SubTokenizer {
    constructor(private readonly isFenceStart: (cursor: TokenizerCursor) => boolean) {}

    match(cursor: TokenizerCursor): boolean {
        // EOFでなければ常にブロックを開始できる（最悪単一のテキストとして処理）
        return !cursor.isEof;
    }

    parse(cursor: TokenizerCursor): BlockToken {
        const startPos = cursor.position;

        while (!cursor.isEof) {
            // 1. 2連続改行が来たら、ブロックの境界なので終了する
            if (cursor.peek(0) === '\n' && cursor.peek(1) === '\n') {
                break;
            }

            // 2. 改行の直後（＝次の行頭）にフェンスが始まるか確認する
            if (cursor.peek(0) === '\n') {
                // 1文字進めて、次の行頭のカーソル状態を仮想的に作る
                cursor.advance(1);
                const isFenceNext = this.isFenceStart(cursor);
                
                // 次の行頭からフェンスが始まるなら、この改行の手前（前の行末）でブロックを終了する
                if (isFenceNext) {
                    // advanceした分を考慮し、改行の手前で止めるためのロジック
                    // 実際には、改行自体はブロックに含めず、外側で制御するか、
                    // ここでループを抜ける。今回は直前の状態を維持するため、
                    // カーソルの位置調整をしてブレイクします。
                    break;

## 1. 複数行の接辞を持つフェンス解析は困難か？
結論から申し上げると、通常の「1字ずつ進む前向き走査（LL(1)等のパーサ）」だけでは極めて困難ですが、「2段階（2パス）の走査」または「バックトラック（巻き戻し）」を導入することで論理的に破綻なく実装可能です。
## 困難とされる理由（論理的説明）
最悪のケースとして、以下のような構造が入力された場合を考えます。

＝＝＝＝＝
A（これが接頭辞なのか、単なるブロックテキストなのか？）
＝＝＝＝＝
B（これがフェンス内容なのか、それとも接尾辞なのか？）
＝＝＝＝＝

1遍の走査（シングルパス）で上から順に処理しようとすると、2つ目の ＝＝＝＝＝ に到達した時点で、それが「接頭辞の終わり」なのか「フェンス全体の終わり」なのかを確定（極定）できません。
## 解決のための論理的アプローチ
この問題を解決するためには、「まず外枠の境界（最も外側のフェンス記号）をすべて特定し、その後に内部を切り分ける」 というアプローチを取ります。

   1. フェンス記号（＝＝＝等）の最外殻のペアを探索する:
   行頭にあるフェンス記号を走査し、同じ長さ、あるいはそれ以上の長さの「閉じフェンス記号」がどこにあるかを、ドキュメント全体または後続行から先に見つけ出します。
   2. 境界の確定後に中身をパースする:
   「ここからここまでが1つの巨大なフェンス構造である」と外枠が確定すれば、その内部にある ＝＝＝＝ などの記号は「接頭辞や接尾辞の区切り」として一意に認識できるようになります。

したがって、責任を適切に分離すれば、将来的にこの拡張が入ってもパーサ全体の基本設計を壊さずに拡張可能です。

## ② 2連続以上の改行（スペース）を処理する責任: SpaceTokenizer
「3連続改行であれば 3 - 2 = 1」の仕様を厳密に計算します。また、フェンスやブロックの境界として機能する「2連続改行」自体は、このトークナイザが消費して区切りとします。

```typescript
import { SubTokenizer, TokenizerCursor } from './tokenizer-interface.js';
import { SpaceToken } from './token.js';

export class SpaceTokenizer implements SubTokenizer {
    match(cursor: TokenizerCursor): boolean {
        // 2連続以上の改行（LF）から始まる場合のみマッチ
        return cursor.peek(0) === '\n' && cursor.peek(1) === '\n';
    }

    parse(cursor: TokenizerCursor): SpaceToken {
        const startPos = cursor.position;
        let lfCount = 0;

        // 連続する改行をすべてカウントして消費する
        while (!cursor.isEof && cursor.peek() === '\n') {
            lfCount++;
            cursor.advance(1);
        }

        const endPos = cursor.position;
        // 仕様: count = 行数（LFの数） - 2
        const tokenCount = lfCount - 2;

        return {
            kind: 'space',
            range: { start: startPos, end: endPos },
            count: tokenCount
        };
    }
}

## ③ フェンス構造を処理する責任: FenceTokenizer
現時点では将来の複雑な接辞解析を「保留」とし、最も単純な「同じ長さのフェンス記号で挟まれた構造」のみを単一責任として実装します。将来拡張する際はこのクラスの内部ロジック、あるいは具象クラスだけを差し替えます。

```typescript
import { SubTokenizer, TokenizerCursor } from './tokenizer-interface.js';
import { FenceToken } from './token.js';

export class FenceTokenizer implements SubTokenizer {
    match(cursor: TokenizerCursor): boolean {
        // フェンスは必ず行頭始まり（今回は簡易的に3連続以上の特定の記号を想定。例として「＝」や「-」）
        const char = cursor.peek(0);
        if (char !== '＝' && char !== '-' && char !== '`') {
            return false;
        }
        return cursor.peek(1) === char && cursor.peek(2) === char;
    }

    parse(cursor: TokenizerCursor): FenceToken {
        const startPos = cursor.position;
        const char = cursor.peek();
        
        // 連続するフェンス記号の数を数える
        let num = 0;
        while (cursor.peek(num) === char) {
            num++;
        }

        // 開始フェンス行の終了（LF）まで進める
        const prefixStart = startPos;
        cursor.advance(num);
        while (!cursor.isEof && cursor.peek() !== '\n') { cursor.advance(1); }
        if (cursor.peek() === '\n') { cursor.advance(1); }
        const prefixEnd = cursor.position;

        const contentStart = cursor.position;
        let contentEnd = cursor.position;
        let suffixStart = cursor.position;
        let suffixEnd = cursor.position;

        // 閉じフェンスの探索
        while (!cursor.isEof) {
            const currentPos = cursor.position;
            
            // 行頭で同じ文字が同じ数以上連続しているか
            let matchCount = 0;
            while (cursor.peek(matchCount) === char) {
                matchCount++;
            }

            if (matchCount >= num) {
                contentEnd = currentPos;
                suffixStart = currentPos;
                cursor.advance(matchCount);
                while (!cursor.isEof && cursor.peek() !== '\n') { cursor.advance(1); }
                if (cursor.peek() === '\n') { cursor.advance(1); }
                suffixEnd = cursor.position;
                break;
            }

            // 次の行へ
            while (!cursor.isEof && cursor.peek() !== '\n') { cursor.advance(1); }
            if (cursor.peek() === '\n') { cursor.advance(1); }
            contentEnd = cursor.position;
        }

        return {
            kind: 'fence',
            range: { start: startPos, end: cursor.position },
            enclosure: { char, num },
            prefix: { range: { start: prefixStart, end: prefixEnd } },
            content: { range: { start: contentStart, end: contentEnd } },
            suffix: { range: { start: suffixStart, end: suffixEnd } }
        };
    }
}
```

## ④ ブロックを処理する責任: BlockTokenizer
「フェンスでもなく、2連続改行（スペース）でもない」テキストの塊を処理します。次のフェンスの開始、または2連続改行が出現した時点で処理の責任を終えます。

```typescript
import { SubTokenizer, TokenizerCursor } from './tokenizer-interface.js';
import { BlockToken } from './token.js';

export class BlockTokenizer implements SubTokenizer {
    constructor(private readonly isFenceStart: (cursor: TokenizerCursor) => boolean) {}

    match(cursor: TokenizerCursor): boolean {
        // EOFでなければ常にブロックを開始できる（最悪単一のテキストとして処理）
        return !cursor.isEof;
    }

    parse(cursor: TokenizerCursor): BlockToken {
        const startPos = cursor.position;

        while (!cursor.isEof) {
            // 1. 2連続改行が来たら、ブロックの境界なので終了する
            if (cursor.peek(0) === '\n' && cursor.peek(1) === '\n') {
                break;
            }

            // 2. 改行の直後（＝次の行頭）にフェンスが始まるか確認する
            if (cursor.peek(0) === '\n') {
                // 1文字進めて、次の行頭のカーソル状態を仮想的に作る
                cursor.advance(1);
                const isFenceNext = this.isFenceStart(cursor);
                
                // 次の行頭からフェンスが始まるなら、この改行の手前（前の行末）でブロックを終了する
                if (isFenceNext) {
                    // advanceした分を考慮し、改行の手前で止めるためのロジック
                    // 実際には、改行自体はブロックに含めず、外側で制御するか、
                    // ここでループを抜ける。今回は直前の状態を維持するため、
                    // カーソルの位置調整をしてブレイクします。
                    break; 
                }
                // フェンスでなければ、改行をブロックの一部として消費して続行
                continue;
            }

            cursor.advance(1);
        }

        return {
            kind: 'block',
            range: { start: startPos, end: cursor.position }
        };
    }
}
```

## 1. ご指摘の論理検証（改行の責任分界点について）
## 改行を誰が消費すべきか
ご指摘の通り、「フェンスやブロックが自身の終端文字である2連続改行を消費すべき」という設計の方が、単一責任の原則においてより堅牢な論理となります。
前回の設計（SpaceTokenizer が2連続改行を消費する）では、Block や Fence は「2連続改行の手前」でパースを中断し、カーソルをその場に残す必要がありました。この場合、Block は「2連続改行によって自分が終わった」というコンテキストを知っているにもかかわらず、その消費を外部に委ねることになり、結果として各トークナイザ間で改行の押し付け合い（結合度の増加）が発生します。
## 新しい論理構造（セパレータによる区切り）

   1. Block や Fence は、自身を成立させる最小単位（構造）のパースを終えた後、その構造を締めくくる「構造区切り（P）」を自身で消費して終了します。
   2. SpaceTokenizer は、構造のパースが終わった「後」のカーソル位置から始まります。そこでさらに「構造区切り（P）」が連続して存在する場合のみ、それを「不要な余白（SpaceToken）」としてカウントして消費します。

これにより、すべての構造トークナイザは「自身の開始から、直後の区切り消費まで」を自己完結（単一責任）できるようになります。
------------------------------
## 2. 接辞トークナイザ（AffixTokenizer）の設計論理
接辞（Prefix/Suffix）は、フェンスの「外殻」の内部に位置する独立したテキスト構造です。フェンス記号と内容（Content）の間に挟まれる性質を持ちます。

* null（接辞なし）: フェンス記号の直後が「行内改行（L）」または「構造区切り（P）」でなければならないという制約（バリデーション）の役割を持ちます。
* OneAffixTokenizer: フェンス記号の直後から「行内改行（L）」または「構造区切り（P）」の手前までを1つの接辞トークンとして切り出します。

これらをフェンスコンストラクタへ依存注入（DI）することで、フェンスのパースアルゴリズムを変更することなく、接辞のパースルールだけを柔軟に切り替えられる（DRYかつ単一責任の）構造を実現します。

```typescript
import { BaseToken, FenceToken, BlockToken, SpaceToken } from './token.js';
import { TokenizerCursor, SubTokenizer } from './tokenizer-interface.js';

/**
 * 区切り文字の判定を抽象化する基底クラス
 */
export abstract class Separator {
    abstract isP(cursor: TokenizerCursor): boolean; // 構造区切り (Paragraph / Part)
    abstract isL(cursor: TokenizerCursor): boolean; // 構造内改行 (Line)
    abstract advanceP(cursor: TokenizerCursor): void;
    abstract advanceL(cursor: TokenizerCursor): void;
}

/**
 * ASCII環境（LF*2 / LF）用のセパレータ実装
 */
export class AsciiSeparator extends Separator {
    isP(cursor: TokenizerCursor): boolean {
        return cursor.peek(0) === '\n' && cursor.peek(1) === '\n';
    }
    isL(cursor: TokenizerCursor): boolean {
        return cursor.peek(0) === '\n';
    }
    advanceP(cursor: TokenizerCursor): void {
        cursor.advance(2);
    }
    advanceL(cursor: TokenizerCursor): void {
        cursor.advance(1);
    }
}

/**
 * Unicode環境（PS / LS）用のセパレータ実装
 * 根拠: Unicode Standard Annex #14 (https://unicode.org) において
 * U+2029 (Paragraph Separator) および U+2028 (Line Separator) が規定されています。
 */
export class UnicodeSeparator extends Separator {
    isP(cursor: TokenizerCursor): boolean {
        return cursor.peek(0) === '\u2029';
    }
    isL(cursor: TokenizerCursor): boolean {
        return cursor.peek(0) === '\u2028';
    }
    advanceP(cursor: TokenizerCursor): void {
        cursor.advance(1);
    }
    advanceL(cursor: TokenizerCursor): void {
        cursor.advance(1);
    }
}

/**
 * 接辞（Prefix/Suffix）用トークンの共通型定義
 */
export interface AffixToken extends BaseToken {
    kind: 'affix';
    value: string;
}

/**
 * 単一の行内文字列を接辞として切り出すトークナイザ
 */
export class OneAffixTokenizer {
    constructor(private readonly separator: Separator) {}

    match(cursor: TokenizerCursor): boolean {
        // 直後が改行（L）または構造区切り（P）でなければマッチ
        return !this.separator.isL(cursor) && !this.separator.isP(cursor) && !cursor.isEof;
    }

    parse(cursor: TokenizerCursor): AffixToken {
        const start = cursor.position;
        while (!cursor.isEof && !this.separator.isL(cursor) && !this.separator.isP(cursor)) {
            cursor.advance(1);
        }
        const end = cursor.position;
        return {
            kind: 'affix',
            range: { start, end },
            value: cursor.substring(start, end)
        };
    }
}

/**
 * 2連続以上の「構造区切り（P）」をパースする責任
 */
export class SpaceTokenizer implements SubTokenizer {
    constructor(private readonly separator: Separator) {}

    match(cursor: TokenizerCursor): boolean {
        // すでに前段のトークンによって1つ目のPは消費されている想定。
        // ここからさらにPが始まる場合のみ、余剰なスペースとして認識する。
        return this.separator.isP(cursor);
    }

    parse(cursor: TokenizerCursor): SpaceToken {
        const start = cursor.position;
        let pCount = 0;

        // 連続する構造区切りをカウントして消費
        while (!cursor.isEof && this.separator.isP(cursor)) {
            pCount++;
            this.separator.advanceP(cursor);
        }

        return {
            kind: 'space',
            range: { start, end: cursor.position },
            count: pCount // 既に1つ消費された後から数えるため、ここでのカウントがそのまま「行数-2」に相当
        };
    }
}

/**
 * フェンス構造をパースする責任
 */
export class FenceTokenizer implements SubTokenizer {
    constructor(
        private readonly separator: Separator,
        private readonly fenceChar: string,
        private readonly minLength: number = 3,
        private readonly prefixTokenizer: OneAffixTokenizer | null = null
    ) {}

    match(cursor: TokenizerCursor): boolean {
        // コンストラクタで指定された文字が規定数以上連続しているか
        let count = 0;
        while (cursor.peek(count) === this.fenceChar) {
            count++;
        }
        return count >= this.minLength;
    }

    parse(cursor: TokenizerCursor): FenceToken {
        const startPos = cursor.position;
        
        let num = 0;
        while (cursor.peek(num) === this.fenceChar) {
            num++;
        }

        const prefixStart = startPos;
        cursor.advance(num);

        // 接辞のパース（DIされたトークナイザを使用）
        let prefixToken: AffixToken | null = null;
        if (this.prefixTokenizer && this.prefixTokenizer.match(cursor)) {
            prefixToken = this.prefixTokenizer.parse(cursor);
        }

        // 接辞の後、行末（L）または構造区切り（P）を消費して1行目を終える
        if (this.separator.isP(cursor)) {
            this.separator.advanceP(cursor);
        } else if (this.separator.isL(cursor)) {
            this.separator.advanceL(cursor);
        }
        const prefixEnd = cursor.position;

        const contentStart = cursor.position;
        let contentEnd = cursor.position;
        let suffixStart = cursor.position;
        let suffixEnd = cursor.position;

        // 閉じフェンスの探索（一番外側の終了記号を探す）
        while (!cursor.isEof) {
            const currentPos = cursor.position;
            
            let matchCount = 0;
            while (cursor.peek(matchCount) === this.fenceChar) {
                matchCount++;
            }

            // 開始フェンスと同じ長さ以上の記号が出現した場合
            if (matchCount >= num) {
                contentEnd = currentPos;
                suffixStart = currentPos;
                cursor.advance(matchCount);

                // 将来のSuffix拡張に備え、行末を消費して確定させる
                if (this.separator.isP(cursor)) {
                    this.separator.advanceP(cursor);
                } else if (this.separator.isL(cursor)) {
                    this.separator.advanceL(cursor);
                }
                suffixEnd = cursor.position;
                break;
            }

            // 次の行へ移動
            if (this.separator.isP(cursor)) {
                this.separator.advanceP(cursor);
            } else if (this.separator.isL(cursor)) {
                this.separator.advanceL(cursor);
            } else {
                cursor.advance(1);
            }
            contentEnd = cursor.position;
        }

        // このフェンス構造自体の末尾にあるべき「構造区切り（P）」を消費して終了する責任
        if (this.separator.isP(cursor)) {
            this.separator.advanceP(cursor);
        }

        return {
            kind: 'fence',
            range: { start: startPos, end: cursor.position },
            enclosure: { char: this.fenceChar, num },
            prefix: { range: { start: prefixStart, end: prefixEnd } },
            content: { range: { start: contentStart, end: contentEnd } },
            suffix: { range: { start: suffixStart, end: suffixEnd } }
        };
    }
}

/**
 * ブロック構造をパースする責任
 */
export class BlockTokenizer implements SubTokenizer {
    constructor(
        private readonly separator: Separator,
        private readonly isFenceStart: (cursor: TokenizerCursor) => boolean
    ) {}

    match(cursor: TokenizerCursor): boolean {
        return !cursor.isEof;
    }

    parse(cursor: TokenizerCursor): BlockToken {
        const startPos = cursor.position;

        while (!cursor.isEof) {
            // 1. 構造区切り（P）が来たら、ブロックの終了条件を満たす
            if (this.separator.isP(cursor)) {
                break;
            }

            // 2. 行内改行（L）が来た場合、次の行頭からフェンスが始まるか検証する
            if (this.separator.isL(cursor)) {
                // 仮想的に1行進めたカーソル状態を想定
                const nextCursor = {
                    ...cursor,
                    position: cursor.position + 1,
                    peek: (offset: number = 0) => cursor.peek(offset + 1)
                } as TokenizerCursor;

                if (this.isFenceStart(nextCursor)) {
                    // 次の行頭からフェンスが始まるなら、現在の行末（Lの手前）でブロックを終了する
                    break;
                }
            }

            cursor.advance(1);
        }

        // ブロック自体の末尾にある「構造区切り（P）」を消費して終了する責任
        if (this.separator.isP(cursor)) {
            this.separator.advanceP(cursor);
        }

        return {
            kind: 'block',
            range: { start: startPos, end: cursor.position }
        };
    }
}

## 1. 概念定義と命名の論理的アプローチ## 最小単位（Inline）の上位抽象概念の命名
「これ以上分解できない最小の文書構造単位」を意味する抽象概念として、最も適切であると考えられる名称は Element（要素） です。
## 語彙の選定論理

* Element（要素）: 集合や全体を構成する、それ以上分割するとその性質を失う「最小の機能的単位」を指します。
* Atom（原子） / Atomic: 分割不可能な最小単位を意味しますが、コンパイラ設計においてはトークンそのものを指すことが多く、抽象構文木（AST）内の「文書の構成部品」としては無機質すぎます。
* Node（節・点）: グラフ構造全般を指す最汎用語（General Term）であり、AstNode ですでに使用しているため、概念の重複（シャドーイング）が発生します。
* Unit（単位）: 測定基準や物理的なまとまり（モジュール）のニュアンスが強く、文書構造の「構成部品」としての意味合いが薄くなります。

## 枠（Frame）と要素（Element）の階層論理
提案された設計に基づき、文書を構成する概念を2つの階層に分離します。

   1. Frame（枠）: 文書の物理的・空間的な境界を定義する構造（Fence, Block, Space）。
   2. Element（要素）: 枠の内部を満たす、意味情報を持った最小の構成単位（Text, Ruby, Emphasis等）。

これにより、AST（抽象構文木）における責務が明確になります。

## 理由と論理的根拠

* 意味と目的の非対称性: Frame（枠）という言葉は、物理的な外枠や境界、レイアウトの箱を強く連想させます。しかし、今回パース対象とする Space（間）は、境界を持たない「余白」そのものであり、Block もまた単なるテキストの塊です。これらを一括りに「枠」と呼ぶのは、言葉の定義として矛盾が生じます。
* 規則に統治された圏（Zone）: 一方で、Zone は「特定の目的、規則、特徴により区切られた圏」を指します。Fence は「ネストや接辞の解釈規則を持つ圏」、Block は「インライン要素の展開規則を持つ圏」、Space は「余白のカウント規則を持つ圏」です。それぞれが異なる目的と規則を持ち、個別のTokenizerを必要とする実情に完全に合致します。

```typescript
import { TokenizerCursor } from './cursor.js';

/**
 * すべての区切り管理者の最上位
 */
export class Separator {
    private readonly _char: string;

    constructor(char: string) {
        this._char = char;
    }

    get char(): string {
        return this._char;
    }

    is(cursor: TokenizerCursor): boolean {
        // substringの引数は(start, end)のため、現在の位置から文字数分を切り出す
        return this._char === cursor.substring(cursor.position, cursor.position + this._char.length);
    }

    advance(cursor: TokenizerCursor): void {
        cursor.advance(this._char.length);
    }
}

/**
 * 複数のSeparatorを動的に管理し、ゲッターを自動生成するクラス
 */
export class MultiSeparator extends Separator {
    // 内部で具象Separatorを保持する辞書
    protected seps: Record<string, Separator> = {};

    constructor(options: Record<string, string>, count: number) {
        // MultiSeparator自体は空の文字を持つダミーとして初期化
        super('');
        this.validate(options, count);
        this.make(options);
        this.makeGetter(options);
    }

    protected validate(options: Record<string, string>, count: number): boolean {
        const keys = Object.keys(options);
        const values = Object.values(options);

        if (count !== keys.length) {
            throw new Error(`長さが違います。期待:${count} 実際:${keys.length}`);
        }

        if (!this.valid(keys)) {
            throw new Error(`key が重複しています。:${keys}`);
        }

        if (!this.valid(values)) {
            throw new Error(`value が重複しています。:${values}`);
        }

        return true;
    }

    private valid(values: string[]): boolean {

``````markdown
## 1. validate メソッドをDRYに記述しなかった理由の論理的説明
前回の実装において key と value の検証を別々に記述し、共通化（DRY化）を怠ったのは当方の設計過失です。
## 理由の分析
keys と values はどちらも「文字列の配列（string[]）」であり、実行する検証内容（重複チェックおよびエラー送出）の論理構造は同一です。これらを共通のループまたは高階関数に集約しなかったため、コードの重複が発生しました。
## 修正方針
検証対象の名称（'key' または 'value'）と、その配列データをタプルとして受け取り、単一の検証ループで処理する部品へと細分化します。
------------------------------
## 2. Separator における空文字の制限と初期化ダミー値の検証## char: string に null は代入禁止か
禁止です。 TypeScriptにおいて char: string と型定義している場合、tsconfig.json の strictNullChecks が有効（推奨設定）であれば、null や undefined の代入はコンパイルエラーとなります。したがって、型定義のレベルで null は安全に排除されます。
## 空文字バリデーションの修正
ご提示いただいた Separator.validate 内の this.char.length は、プロパティ代入前に評価されるため、引数の char.length を参照するように修正する必要があります。
## MultiSeparator から super へ渡す初期化値の論理
MultiSeparator は単一のメタ文字を持たず、内部の seps で複数のメタ文字を管理する責任を持ちます。しかし、Separator を継承しているため、親クラスのコンストラクタ（super）を呼び出す必要があります。
空文字を例外とするバリデーションを追加した場合、super('') はエラーになります。これに対する解決策として、'default-dummy-value' のような仮の文字列を渡すアプローチは、将来的にその文字列が偶然マッチしてしまうリスク（論理的脆弱性）を孕みます。
最も堅牢な解決策は、MultiSeparator では親クラスの is や advance メソッドを完全にオーバーライドして無効化（あるいは例外送出）することです。これにより、ダミー値がパースに干渉する危険性を論理的に排除できます。
------------------------------
## 3. サロゲートペア・書記素（Grapheme）対応とパフォーマンスのトレードオフ
十万字規模の原稿を処理する場合、メモリ消費と処理速度のバランスが極めて重要になります。
## 2つの選択肢の論理的検証## 選択肢A: charAt のまま進める

* メリット: 高速であり、メモリ消費は原稿の文字列実体（1オブジェクト）のみです。
* デメリット: メタ文字に絵文字や一部の希少漢字（例：𠮷など）が指定された場合、サロゲートペアの上位または下位の16ビット（コードユニット）のみを誤って切り出すため、正確にマッチしません。

## 選択肢B: Intl.Segmenter や Array.from で事前に全分割する

* メリット: 自由なメタ文字設計が破綻なく成立します。
* デメリット: 文字列を配列化すると、各要素への参照オーバーヘッドと配列自体のメモリが追加され、十万字規模では数メガバイト以上のメモリを余分に消費します（単純計算で元の文字列の数倍のメモリ負荷）。

## 提案：第3の論理的アプローチ（オンデマンドなコードポイント走査）
メモリを倍消費する事前割当（全分割配列化）を避けつつ、自由なUnicode一文字（コードポイント）を安全に扱うためには、「文字列データは string のまま保持し、カーソルの移動（advance）時のみ String.prototype.codePointAt() を使用して可変長で進める」 という設計が最適です。

* codePointAt() を使用すると、その位置にある文字がサロゲートペア（2つのコードユニットで1文字）であれば 0x10000 以上の値を返し、通常の文字であれば 0xffff 以下の値を返します。
* 1文字進める際、その文字がサロゲートペアであればインデックスを +2 し、そうでなければ +1 します。
* この方式であれば、メモリ消費は string 1つ分のまま据え置きとなり、十万字規模でもメモリ負担は発生しません。処理速度への影響も最小限に抑えられます。
* 根拠：[MDN Web Docs: String.prototype.codePointAt()](https://developer.mozilla.org/ja/docs/Web/JavaScript/Reference/Global_Objects/String/charAt)

## tokenizer/core/cursor.ts
メモリ二重管理を回避しつつ、Unicodeコードポイント単位で正確に文字を進める責任を持つ StringCursor です。

```typescript
export interface TokenizerCursor {
    readonly text: string;
    readonly position: number;
    readonly isEof: boolean;
    peek(offset?: number): string;
    advance(count?: number): void;
    advancePoint(): void; // Unicode 1コードポイント分進める
    substring(start: number, end: number): string;
}

export class StringCursor implements TokenizerCursor {
    private index: number = 0;
    constructor(public readonly text: string) {}

    get position(): number { return this.index; }
    get isEof(): boolean { return this.index >= this.text.length; }

    peek(offset: number = 0): string {
        return this.text.charAt(this.index + offset);
    }

    advance(count: number = 1): void {
        this.index += count;
    }

    /**
     * 現在位置の文字がサロゲートペアかどうかを判定し、正確に1文字分インデックスを進める責任
     */
    advancePoint(): void {
        if (this.isEof) return;
        
        const code = this.text.codePointAt(this.index);
        // codePointAt が 0xFFFF を超える場合はサロゲートペア（2コードユニット消費）
        if (code !== undefined && code > 0xFFFF) {
            this.index += 2;
        } else {
            this.index += 1;
        }
    }

    substring(start: number, end: number): string {
        return this.text.substring(start, end);
    }
}
```

        // バリデーション回避用の固定ダミー文字列。

        this.make(options);
        this.makeGetter(options);

     * key と value の重複検証をDRYに実行する責任
     */
    protected validateOptions(options: Record<string, string>, count: number): boolean {
        const keys = Object.keys(options);
        const values = Object.values(options);

        if (count !== keys.length) {
            throw new Error(`長さが違います。期待:${count} 実際:${keys.length}`);
        }

        // 検証対象の名称と配列をタプルにし、同一の論理ループでDRYに検証
        const validationTargets: [string, string[]][] = [
            ['key', keys],
            ['value', values]
        ];

        for (const [name, targetArray] of validationTargets) {
            if (!this.valid(targetArray)) {
                throw new Error(`${name} が重複しています。:${targetArray}`);
            }
        }

        return true;
    }

    private valid(values: string[]): boolean {
        return values.length === (new Set(values)).size;
    }

    protected make(options: Record<string, string>): void {
        for (const [k, v] of Object.entries(options)) {
            this.seps[k] = new Separator(v);
        }
    }

    protected makeGetter(options: Record<string, string>): void {
        for (const key of Object.keys(options)) {
            Object.defineProperty(this, key, {
                get: () => this.seps[key],
                enumerable: true,
                configurable: false
            });
        }
    }

    /**
     * 親クラスの単一文字判定ロジックが誤って動かないよう、明示的に上書きして無効化する責任

    }

## 1. JavaScriptの文字列、サロゲートペア、Unicode仕様の論理的説明
混乱を防ぐため、JavaScriptが扱う「文字の単位」に関する事実と論理を、公式仕様に基づいて整理します。
## ① なぜ +2 で固定なのか？（UTF-16の仕様）
結論から申し上げますと、codePointAt() が 0xFFFF を超えたときに進めるべきインデックス数が +2 固定であることは、UTF-16の規格により厳密に定められています。 +3 や +4 になることは論理的にあり得ません。

* 根拠: ECMA-262 Standard (ECMAScript Language Specification) の「6.1.4 The String Type」において、JavaScriptの文字列は「16ビット符号なし整数値（コードユニット）のシーケンス」であり、UTF-16符号化形式を使用することが明記されています。
ECMAScript Specification: The String Type
* 論理: Unicodeの全コードポイント（U+0000 〜 U+10FFFF）のうち、16ビット（0xFFFF）に収まらない文字（U+10000以降の文字）は、UTF-16においては必ず「上位サロゲート（16ビット）」と「下位サロゲート（16ビット）」の合計32ビット（2コードユニット）のペアに変換して保持されます。これより長い3コードユニット（48ビット）や4コードユニット（64ビット）で1つのコードポイントを表現する仕組みは、UTF-16の仕様（規格）自体に存在しません。そのため、codePointAt() が 0xFFFF を超えるコードポイントを返したとき、メモリ上で消費している領域は「絶対に2（16ビット×2）」です。

## ② 合字（Grapheme Cluster）との混同について
「実質無制限にバイナリをくっつけることができる仕様」というのは、1つのコードポイントの長さではなく、「複数のコードポイントを結合させて1つの見た目の文字（書記素：Grapheme Cluster）を作る仕様」 のことです。

* 具体例（家族の絵文字 👨‍👩‍👧）:
この絵文字は画面上では1文字に見えますが、内部は以下の5つのコードポイント（計7つのUTF-16コードユニット） が結合コントロール文字（ZWJ: Zero Width Joiner）で連結されたものです。
1. 👨 (男性: U+1F468 -> サロゲートペアのため2ユニット)
   2. (ZWJ) (結合文字: U+200D -> 1ユニット)
   3. 👩 (女性: U+1F469 -> サロゲートペアのため2ユニット)
   4. (ZWJ) (結合文字: U+200D -> 1ユニット)
   5. 👧 (女の子: U+1F467 -> サロゲートペアのため2ユニット)
* 論理: codePointAt() は、この結合された塊の「先頭にある1つのコードポイント（この場合は 👨）」しか読み取りません。そのため、codePointAt() の戻り値が 0xFFFFFFFF のような巨大な値になることはなく、Unicodeの最大値である 0x10FFFF（10進数で1,114,111）を超えることはありません。

## ③ 単位・APIの正確な一覧表
これまでの混乱を解消するため、正しい対応関係を以下に定義します。

| 階層・単位 | 対応する呼称 | JavaScriptのAPI | 概要 |
|---|---|---|---|
| CodeUnit (16bit) | コードユニット | charAt(), charCodeAt() | JavaScriptのインデックスの基本単位。サロゲートペアを考慮しないため、文字の泣き別れが発生する。 |
| CodePoint (Unicode1字) | コードポイント | codePointAt(), String.fromCodePoint() | Unicodeで定義された一連の文字の識別番号。サロゲートペアを正しく1文字として扱う最小の「意味単位」。 |
| Grapheme (書記素/見た目の1字) | 書記素クラスター | Intl.Segmenter | 結合絵文字や濁点結合（か＋゙＝が）など、人間が見たときに「1文字」と認識する最終的な文字の塊。 |

私たちが作っている文書構造パーサ（Tokenizer）において、メタ文字として機能させたい最小の単位は、CodeUnit（不完全なバイナリ）ではなく、CodePoint（Unicode 1文字単位） です。

## 理由の分析

## 3. 単一責任の原則（SRP）を極限まで死守するクラス分割設計
「状況に応じて文字を扱う単位を切り替える」「責任分界点を極限まで細分化する」という絶対命令に基づき、カーソルクラスの責務を3つの独立したクラスへ完全に分離（細分化）します。

   1. CodeUnitCursor: JavaScript標準の16bit単位（CodeUnit）でのみ動作する、高速かつ最も低レイヤな移動の責任。
   2. CodePointCursor: サロゲートペアの仕様（U+10FFFF）を内部に閉じ込め、外部には「正確なUnicode 1文字単位」での移動・先読みを提供する責任。

```typescript
/**
 * 16ビット（UTF-16コードユニット）単位の操作に特化した最小責任クラス
 */
export class CodeUnitCursor {
    private idx: number = 0;

    constructor(private readonly text: string) {}

    get position(): number {
        return this.idx;
    }

    set position(pos: number) {
        this.idx = pos;
    }

    get isEof(): boolean {
        return this.idx >= this.text.length;
    }

    /** 16bitユニット単体の数値を読み取る */
    codeAt(offset: number = 0): number {
        return this.text.charCodeAt(this.idx + offset);
    }

    /** 指定された16bitユニット数分だけインデックスを愚直に進める */
    advanceUnit(count: number): void {
        this.idx += count;
    }
}

/**
 * Unicodeコードポイント（サロゲートペア対応）単位の計算に特化した責任クラス
 */
export class CodePointCursor {
    private readonly unitCursor: CodeUnitCursor;

    constructor(private readonly text: string) {
        this.unitCursor = new CodeUnitCursor(text);
    }

    get isEof(): boolean {
        return this.unitCursor.isEof;
    }

    /** 現在の物理的なUTF-16インデックス位置を取得する（Tokenのrange記録用） */
    getAbsolutePosition(): number {
        return this.unitCursor.position;
    }

    /** 現在位置から、指定されたコードポイント（文字数）分だけ進んだ「物理インデックス」を計算して返す責任 */
    private calculateTargetUnitPosition(charOffset: number): number {
        // 現在のカーソル位置を一時的にコピー（退避）して計算を行う
        const currentPos = this.unitCursor.position;
        const tempCursor = new CodeUnitCursor(this.text);
        tempCursor.position = currentPos;

        let charCount = 0;
        while (charCount < charOffset && !tempCursor.isEof) {
            const code = this.text.codePointAt(tempCursor.position);
            
            // UTF-16の仕様: 0xFFFFを超えていればサロゲートペアなので2ユニット、それ以外は1ユニット消費
            if (code !== undefined && code > 0xFFFF) {
                tempCursor.advanceUnit(2);
            } else {
                tempCursor.advanceUnit(1);
            }
            charCount++;
        }

        const targetPos = tempCursor.position;
        return targetPos;
    }

    /** 指定された文字数（コードポイント単位）先の数値を取得する */
    peekCode(charOffset: number = 0): number | undefined {
        if (this.isEof) return undefined;
        const targetUnitPos = this.calculateTargetUnitPosition(charOffset);
        if (targetUnitPos >= this.text.length) return undefined;
        return this.text.codePointAt(targetUnitPos);
    }

    /** 指定された文字数（コードポイント単位）だけ実際にカーソルを進行させる */
    advancePoint(charCount: number = 1): void {
        const nextUnitPos = this.calculateTargetUnitPosition(charCount);
        this.unitCursor.position = nextUnitPos;
    }
}

/**
 * トークナイザが利用する高レイヤな抽象インターフェース
 */
export interface TokenizerCursor {
    readonly isEof: boolean;
    peekChar(offset?: number): string;
    peekCode(offset?: number): number | undefined;
    advance(charCount?: number): void;
    substring(start: number, end: number): string;
    getAbsolutePosition(): number;
}

/**
 * CodePointCursorの計算ロジックを利用し、トークナイザ向けの文字列操作を提供する責任クラス
 */
export class StringCursor implements TokenizerCursor {
    private readonly cpCursor: CodePointCursor;

    constructor(private readonly text: string) {
        this.cpCursor = new CodePointCursor(text);
    }

    get isEof(): boolean {
        return this.cpCursor.isEof;
    }

    getAbsolutePosition(): number {
        return this.cpCursor.getAbsolutePosition();
    }

    peekCode(offset: number = 0): number | undefined {
        return this.cpCursor.peekCode(offset);
    }

    peekChar(offset: number = 0): string {
        const code = this.cpCursor.peekCode(offset);
        if (code === undefined) return '';
        return String.fromCodePoint(code);
    }

    advance(charCount: number = 1): void {
        this.cpCursor.advancePoint(charCount);
    }

    substring(start: number, end: number): string {
        return this.text.substring(start, end);
    }
}

## 判断の論理的根拠

* 結合コントロール文字の性質: 結合絵文字（例: 👨‍👩‍👧）や結合濁点（例: か ＋ ゙）のような書記素は、テキストの表現力を豊かにするための仕組みです。これらを「フェンスの記号（例: ＝＝＝）」などの構造区切り文字として人間が自ら進んで定義・入力する可能性は極めて低いという予測に基づいています。
* ASCII記号への無意識の最適化: Markdownの ` や -、あるいは今回想定されている ＝ など、一般的なメタ文字はすべて「単一のCodePoint」で完結します。当方はこの前提に無意識に囚われていたため、書記素クラスター全体の解析を過剰な設計とみなし、CodePointまでで十分であるという結論を急ぎました。

## 仕様に照らし合わせた場合の「書記素（Grapheme）単位」の正当性
しかし、ご提示いただいた「どんな字でもメタ文字として受け入れる仕様。つまり人が1字として認識できるすべての字」という設計思想に照らし合わせると、当方の判断は誤りであり、書記素（Grapheme Cluster）単位でのパースが正当となります。
例えば、ユーザーが「家族の絵文字 👨‍👩‍👧」をフェンスのメタ文字として指定し、👨‍👩‍👧👨‍👩‍👧👨‍👩‍👧（3連続）でフェンスを開始したいという構文ルールを設計した場合、CodePoint単位のカーソルでは 👨（男性）だけを読み取ってしまい、後続の結合文字（ZWJ等）を取り残して構文解析が崩壊します。
したがって、「任意の文字をメタ文字にできる自由度」を保証するためには、最上位のカーソルは 書記素（Grapheme Cluster）単位で駆動するべき です。
------------------------------
## 4. 単一責任の原則（SRP）を死守した3階層カーソルの設計
3つの単位（CodeUnit, CodePoint, Grapheme）の責任を極限まで細分化し、それぞれが単一の関心事のみを処理する独立したクラスとして再構築します。

   1. CodeUnitCursor: 16ビット（UTF-16）の物理インデックスを愚直に進める責任。
   2. CodePointCursor: サロゲートペア（0xFFFF超え）の長さを計算し、Unicodeの符号位置単位で進める責任。
   3. GraphemeCursor: JavaScript標準の Intl.Segmenter のロジックをカプセル化し、人間の視認できる1字（書記素）の長さを計算して進める責任。
   4. StringCursor: GraphemeCursor の責任にのみ依存し、トークナイザに安全な文字列操作（peek, advance）を提供する責任。

## Intl.Segmenter の採用とメモリ負荷に対する論理的解決
以前、Intl.Segmenter の使用によりメモリが倍消費される懸念について触れました。

* 根拠: Intl.Segmenter で Array.from(segmenter.segment(text)) のように一度に全展開して配列化すると、文字列データの二重保持となりメモリを大量に消費します（MDN Web Docs: Intl.Segmenter）。
* 解決策: 事前に全展開せず、「現在位置から次の1文字が何バイト（何ユニット）あるか」をその都度オンデマンドに segmenter.segment(text) を用いてイテレータとして部分走査する 仕組みにします。これにより、メモリの消費量は原稿の文字列1つ分（string）のままに抑えられ、十万字規模の処理におけるメモリ負荷を論理的に回避できます。

## 5. 3単位すべてを網羅したリファクタリングコード## tokenizer/core/cursor.ts

```typescript
import { TokenizerCursor } from './cursor.js';

/**
 * 【第1階層】16ビット（UTF-16コードユニット）単位の操作に特化した最小責任クラス
 */
export class CodeUnitCursor {
    private idx: number = 0;

    constructor(private readonly text: string) {}

    get position(): number {
        return this.idx;
    }

    set position(pos: number) {
        this.idx = pos;
    }

    get isEof(): boolean {
        return this.idx >= this.text.length;
    }

    advanceUnit(count: number): void {
        this.idx += count;
    }
}

/**
 * 【第2階層】Unicodeコードポイント（サロゲートペア対応）単位の計算に特化した責任クラス
 */
export class CodePointCursor {
    private readonly unitCursor: CodeUnitCursor;

    constructor(private readonly text: string) {
        this.unitCursor = new CodeUnitCursor(text);
    }

    get isEof(): boolean {
        return this.unitCursor.isEof;
    }

    getAbsolutePosition(): number {
        return this.unitCursor.position;
    }

    setAbsolutePosition(pos: number): void {
        this.unitCursor.position = pos;
    }

    /** 現在位置から、指定されたコードポイント数分進んだ物理インデックスを計算する */
    calculateTargetUnitPosition(charOffset: number): number {
        const tempCursor = new CodeUnitCursor(this.text);
        tempCursor.position = this.unitCursor.position;

        let charCount = 0;
        while (charCount < charOffset && !tempCursor.isEof) {
            const code = this.text.codePointAt(tempCursor.position);
            if (code !== undefined && code > 0xFFFF) {
                tempCursor.advanceUnit(2); // サロゲートペア
            } else {
                tempCursor.advanceUnit(1); // 通常文字
            }
            charCount++;
        }
        return tempCursor.position;
    }

    advancePoint(charCount: number = 1): void {
        this.unitCursor.position = this.calculateTargetUnitPosition(charCount);
    }
}

/**
 * 【第3階層】書記素クラスター（人間が見た目の1字と認識する単位）の計算に特化した責任クラス
 * 根拠: ECMAScript国際化API仕様 (https://tc39.es) 
 * Intl.Segmenter は指定された粒度（granularity: 'grapheme'）で文字列を正確に分割します。
 */
export class GraphemeCursor {
    private readonly cpCursor: CodePointCursor;
    private readonly segmenter: Intl.Segmenter;

    constructor(private readonly text: string) {
        this.cpCursor = new CodePointCursor(text);
        this.segmenter = new Intl.Segmenter(undefined, { granularity: 'grapheme' });
    }

    get isEof(): boolean {
        return this.cpCursor.isEof;
    }

    getAbsolutePosition(): number {
        return this.cpCursor.getAbsolutePosition();
    }

    /** 現在位置から、指定された書記素（見た目の文字数）分進んだ物理インデックスをオンデマンドに計算する */
    calculateTargetGraphemePosition(graphemeOffset: number): number {
        if (graphemeOffset === 0) {
            return this.cpCursor.getAbsolutePosition();
        }

        // オンデマンド走査: 現在位置以降のテキストの一部分だけをSegmenterに投入してメモリ消費を抑える
        const currentPos = this.cpCursor.getAbsolutePosition();
        const segments = this.segmenter.segment(this.text.substring(currentPos));
        
        let count = 0;
        let relativeIndex = 0;

        for (const segment of segments) {
            if (count === graphemeOffset) {
                break;
            }
            // 各セグメント（書記素）の文字列の長さ（CodeUnit数）を取得
            relativeIndex += segment.segment.length;
            count++;
        }

        return currentPos + relativeIndex;
    }

    advanceGrapheme(graphemeCount: number = 1): void {
        const nextPos = this.calculateTargetGraphemePosition(graphemeCount);
        this.cpCursor.setAbsolutePosition(nextPos);
    }
}

/**
 * トークナイザに提供する共通抽象インターフェース
 */
export interface TokenizerCursor {
    readonly isEof: boolean;
    peekChar(offset?: number): string;
    advance(graphemeCount?: number): void;
    substring(start: number, end: number): string;
    getAbsolutePosition(): number;
}

/**
 * 【統括】GraphemeCursorを利用し、トークナイザへ見た目の1字単位の操作を提供する責任クラス
 */
export class StringCursor implements TokenizerCursor {
    private readonly graphCursor: GraphemeCursor;

    constructor(private readonly text: string) {
        this.graphCursor = new GraphemeCursor(text);
    }

    get isEof(): boolean {
        return this.graphCursor.isEof;
    }

    getAbsolutePosition(): number {
        return this.graphCursor.getAbsolutePosition();
    }

    peekChar(offset: number = 0): string {
        if (this.isEof) return '';
        const start = this.graphCursor.calculateTargetGraphemePosition(offset);
        const end = this.graphCursor.calculateTargetGraphemePosition(offset + 1);
        return this.text.substring(start, end);
    }

    advance(graphemeCount: number = 1): void {
        this.graphCursor.advanceGrapheme(graphemeCount);
    }

    substring(start: number, end: number): string {
        return this.text.substring(start, end);
    }
}

## 1. 「メタ文字から最適なカーソルを自動選択する」設計の論理的検証
ご提案いただいた「使用するメタ文字を事前に収集し、その特性（ASCII / サロゲートペア / 結合文字など）に応じて最適なカーソルを自動選択する（CharCursorSelector）」という設計思想は、十万字規模の原稿を高速にパースするための最適化アプローチとして非常に論理的であり、極めて妥当です。
## 利点と必然性の論理

* 圧倒的なコスト削減: ご指摘の通り、Intl.Segmenter（国際化API）のインスタンス化やオンデマンドの文字列セグメンテーションは、V8などのJavaScriptエンジン内部で複雑なロケール判定やUnicodeデータベース参照を伴うため、ASCII文字を走査するだけの用途には過剰であり、無視できない処理オーバーヘッドになります。
* 安全性と速度の両立: メタ文字に絵文字や結合文字が含まれていないことが保証されていれば、低レイヤな CodeUnitCursor や CodePointCursor まで処理をフォールバック（格下げ）させても、構文解析が崩壊（泣き別れ）するリスクは論理的に 0% です。

## 懸念点：原稿の「地の文（テキスト）」に対する泣き別れリスク
この設計を実装するにあたり、1点だけ論理的に回避しなければならない境界条件（バグの引き金）があります。それは「メタ文字そのもの」ではなく、パースされる原稿の「地の文（通常のテキスト）」にサロゲートペアや絵文字が含まれていた場合の挙動です。
例えば、メタ文字にはASCII文字（` や -）しか使われていないため、Selectorが最も高速な CodeUnitCursor を選択したとします。しかし、原稿のブロックテキスト（地の文）の中に「𠮷（サロゲートペア）」や「👨‍👩‍👧（結合絵文字）」が含まれていた場合、どうなるでしょうか。

   1. BlockTokenizer は、次の行頭にフェンス記号（`）が来るまで、1インデックスずつカーソルを進めます。
   2. カーソルの単位が CodeUnit（16bit）のままだと、「𠮷」の前半16bitと後半16bitを別々の2文字としてスキャンすることになります。
   3. 今回の仕様では、ブロックの終了条件は「構造区切り（P）」または「次の行頭のフェンス記号」であるため、地の文の文字が泣き別れになっても、偶然その断片がフェンス記号（`）と一致しない限り、トークナイザの境界判定（フェンスかブロックか）自体はバグらずに正常終了します。
   4. 切り出された BlockToken の範囲（LinerRange）も、インデックスの整合性は保たれているため、後段の Lexer にそのまま正しい位置情報を渡せます。

したがって、「トークナイザが探すべき境界標識（メタ文字）に合字やサロゲートペアが含まれていないなら、地の文にそれらが含まれていても、トークナイザ層は CodeUnit 駆動でバグる事なく最速で動作する」 という論理が成立します。このため、メタ文字だけを収集して選択するアプローチで全く問題ありません。

## 2. 責任分界点の細分化：全メタ文字を収集する責任者（Registry）
この設計思想を単一責任の原則（SRP）に基づいて極限まで細分化するため、以下の3つのコンポーネントに分離します。

   1. MetaCharRegistry（収集・参照の責任）:
   Separator や 各 ZoneTokenizer が使用するすべてのメタ文字（文字列）を一元的に登録・蓄積し、ユニークな一覧として提供する責任を持ちます。
   2. CharCursorSelector（判断・選択の責任）:
   MetaCharRegistry からメタ文字一覧を受け取り、それらの「最大文字長」や「Unicode特性」を検証して、どのカーソルが最適かを決定する責任を持ちます。
   3. CharCursor 階層（具象操作の責任）:
   選択された単位に基づいて、実際にカーソルを駆動する共通のインターフェースです。

## tokenizer/core/cursor.ts
3つのカーソルを CharCursor という共通の抽象インターフェースで統括します。

```typescript
/**
 * トークナイザが利用する高レイヤな文字操作の共通契約（インターフェース）
 */
export interface CharCursor {
    readonly isEof: boolean;
    peekChar(offset?: number): string;
    advance(charCount?: number): void;
    getAbsolutePosition(): number;
}

/**
 * 16ビット（UTF-16コードユニット）単位で愚直に駆動する、最速のカーソル責任
 */
export class CodeUnitCursor implements CharCursor {
    private idx: number = 0;
    constructor(private readonly text: string) {}

    get isEof(): boolean { return this.idx >= this.text.length; }
    getAbsolutePosition(): number { return this.idx; }

    peekChar(offset: number = 0): string {
        if (this.isEof) return '';
        const target = this.idx + offset;
        if (target >= this.text.length) return '';
        return this.text.charAt(target);
    }

    advance(charCount: number = 1): void {
        this.idx += charCount;
    }
}

/**
 * Unicodeコードポイント（サロゲートペア対応）単位で駆動する、中速のカーソル責任
 */
export class CodePointCursor implements CharCursor {
    private idx: number = 0;
    constructor(private readonly text: string) {}

    get isEof(): boolean { return this.idx >= this.text.length; }
    getAbsolutePosition(): number { return this.idx; }

    private calculateTargetPosition(charOffset: number): number {
        let tempIdx = this.idx;
        let count = 0;
        while (count < charOffset && tempIdx < this.text.length) {
            const code = this.text.codePointAt(tempIdx);
            tempIdx += (code !== undefined && code > 0xFFFF) ? 2 : 1;
            count++;
        }
        return tempIdx;
    }

    peekChar(offset: number = 0): string {
        if (this.isEof) return '';
        const targetIdx = this.calculateTargetPosition(offset);
        if (targetIdx >= this.text.length) return '';
        const code = this.text.codePointAt(targetIdx);
        return code !== undefined ? String.fromCodePoint(code) : '';
    }

    advance(charCount: number = 1): void {
        this.idx = this.calculateTargetPosition(charCount);
    }
}

/**
 * 書記素クラスター（人間が見た目の1字と認識する単位）で駆動する、安全だが低速なカーソル責任
 */
export class GraphemeCursor implements CharCursor {
    private idx: number = 0;
    private readonly segmenter: Intl.Segmenter;

    constructor(private readonly text: string) {
        this.segmenter = new Intl.Segmenter(undefined, { granularity: 'grapheme' });
    }

    get isEof(): boolean { return this.idx >= this.text.length; }
    getAbsolutePosition(): number { return this.idx; }

    private calculateTargetPosition(graphemeOffset: number): number {
        if (graphemeOffset === 0) return this.idx;
        const segments = this.segmenter.segment(this.text.substring(this.idx));
        let count = 0;
        let relativeIndex = 0;
        for (const segment of segments) {
            if (count === graphemeOffset) break;
            relativeIndex += segment.segment.length;
            count++;
        }
        return this.idx + relativeIndex;
    }

    peekChar(offset: number = 0): string {
        if (this.isEof) return '';
        const start = this.calculateTargetPosition(offset);
        const end = this.calculateTargetPosition(offset + 1);
        return this.text.substring(start, end);
    }

    advance(charCount: number = 1): void {
        this.idx = this.calculateTargetPosition(charCount);
    }
}

/**
 * 外部の配列やロジックとの互換性、および範囲切り出しを統括するTokenizer専用ラッパー
 */
export class TokenizerCursor implements CharCursor {
    constructor(
        private readonly text: string,
        private readonly cursorImpl: CharCursor
    ) {}

    get isEof(): boolean { return this.cursorImpl.isEof; }
    getAbsolutePosition(): number { return this.cursorImpl.getAbsolutePosition(); }
    peekChar(offset: number = 0): string { return this.cursorImpl.peekChar(offset); }
    advance(charCount: number = 1): void { this.cursorImpl.advance(charCount); }

    substring(start: number, end: number): string {
        return this.text.substring(start, end);
    }
}
```

## tokenizer/core/char-cursor-selector.ts
メタ文字を収集する責任者と、最適なカーソルを選択する責任者の一連の実装です。

```typescript
import { CharCursor, CodeUnitCursor, CodePointCursor, GraphemeCursor, TokenizerCursor } from './cursor.js';

/**
 * すべてのトークナイザ構成部品からメタ文字を一元的に収拾・管理する責任
 */
export class MetaCharRegistry {
    private readonly chars: Set<string> = new Set();

    /** メタ文字（または区切り文字列）をレジストリに登録する */
    register(metaChar: string): void {
        if (metaChar.length === 0) return;
        this.chars.add(metaChar);
    }

    /** 登録されたすべてのユニークなメタ文字のリストを返す */
    getRegisteredChars(): string[] {
        return Array.from(this.chars);
    }
}

/**
 * 収集されたメタ文字のUnicode特性を分析し、最速のカーソルを選択する責任
 */
export class CharCursorSelector {
    private readonly segmenter: Intl.Segmenter;

    constructor() {
        this.segmenter = new Intl.Segmenter(undefined, { granularity: 'grapheme' });
    }

    /**
     * メタ文字の一覧から、安全性とパフォーマンスのトレードオフを計算し、最適なカーソルを生成して返す
     */
    select(text: string, registry: MetaCharRegistry): TokenizerCursor {
        const metaChars = registry.getRegisteredChars();
        
        let requireGrapheme = false;
        let requireCodePoint = false;

        for (const char of metaChars) {
            // 1. 書記素クラスターに分解して「見た目の1字」より内部データ長が長いか調べる
            const segments = Array.from(this.segmenter.segment(char));
            for (const segment of segments) {
                const str = segment.segment;
                
                // 見た目の1字の中に、複数のコードポイント（ZWJ合字など）が含まれている場合
                if (Array.from(str).length > 1) {
                    requireGrapheme = true;
                    break;
                }
                
                // 見た目の1字の内部CodeUnit長が 2（サロゲートペア）である場合
                if (str.length > 1) {
                    requireCodePoint = true;
                }
            }
            if (requireGrapheme) break;
        }

        // 判定論理に基づくカーソル割当
        let selectedImpl: CharCursor;
        if (requireGrapheme) {
            selectedImpl = new GraphemeCursor(text);
        } else if (requireCodePoint) {
            selectedImpl = new CodePointCursor(text);
        } else {
            // すべてのメタ文字が1コードユニット（ASCIIなど）で完結している場合、最速のカーソルを採用
            selectedImpl = new CodeUnitCursor(text);
        }

        return new TokenizerCursor(text, selectedImpl);
    }
}

ご質問の「ASCIIや絵文字（サロゲートペア）が、偶然別のサロゲートペアの断片と一致してしまうか？」という点について、Unicode（UTF-16）の仕様から、「理論上、一致する確率は絶対にゼロ％である」 という決定的な事実をご説明します。
前回の説明で、私はサロゲートペアの範囲を提示しましたが、その本質的な「隔離の論理」を見落としていました。

* 事実（仕様）: Unicode規格において、上位サロゲート（0xD800〜0xDBFF）と下位サロゲート（0xDC00〜0xDFFF）の合計2,048個のコードポイントは、サロゲートペア（32ビット文字）を表現するためだけに完全に専用予約された領域です。

* 論理: すべての通常の文字（ASCII文字、日本語のひらがな・カタカナ・漢字など）は、このサロゲート範囲（0xD800〜0xDFFF）を絶対に使いません。

## この事実から導き出される衝突の論理

   1. ASCIIメタ文字（例: - や `）の場合:
   ASCII文字のコード（0x002D や 0x0060）は、サロゲート範囲と完全に離れています。そのため、原稿の地の文にどれだけ大量の絵文字やサロゲートペア（例: 𠮷）が含まれていても、その内部バイナリ（0xD842 や 0xDFB7）が ASCIIメタ文字と偶然一致することは確率論ではなく、論理的に 100% あり得ません。
   2. 日本語メタ文字（例: ＝）の場合:
   全角の ＝（U+FF1D）もサロゲート範囲外です。地の文のサロゲートペアの断片と一致することは絶対にありません。
   3. 絵文字メタ文字（例: ⭐ U+2B50 や サロゲートペアな文字）の場合:
   メタ文字自体がサロゲートペア（32ビット）の場合であっても、上位サロゲートは上位サロゲートの範囲内、下位は下位の範囲内でしか存在しないため、他の文字の「前半」や「後半」と辻褄が合わない形でクロスして一致することはありません。

## 唯一バグが発生する「真の例外条件」
CodeUnitCursor や CodePointCursor を使ったときに誤認（偶然の一致）が起きる唯一のパターンは、「人間が、意図的に『上位サロゲート単体（不完全なバイナリ文字）』をメタ文字として登録したとき」 だけです。そのような文字はテキストエディタで入力することすら不可能なため、現実の運用では無視して差し支えありません。
------------------------------
## 3. 当方からの新しい解決案：オンデマンド・ハイブリッドカーソル
上記を踏まえ、「原稿の事前スキャン（前払い計算）を一切行わず、メタ文字の登録情報だけで、論理的破綻をゼロにしつつ最速駆動する」 新しいアプローチを提案します。
検証の負担をなくすため、判断基準を「原稿の中身」ではなく、「登録されたメタ文字そのものの構造」 だけに絞り込みます。
## カーソル選択の確定論理

* 条件1（最速・CodeUnit）:
登録されたすべてのメタ文字が「1つのCodeUnit（ASCIIなど）」で構成されている場合。
👉 CodeUnitCursor を選択。 地の文にサロゲートペアがあっても、上述の通り隔離されているため誤認識は絶対に起きません。最速です。
* 条件2（中速・CodePoint）:
登録されたメタ文字に「サロゲートペア（2ユニット文字、漢字や一部の記号など）」が含まれているが、「結合文字（ZWJなどを使った合字）」は含まれていない場合。
👉 CodePointCursor を選択。 地の文に合字（書記素）があっても、メタ文字が合字でないなら、合字のパーツ（ベース文字）にヒットした時点でパース境界として正しく機能するため、誤認は起きません。
* 条件3（安全・Grapheme）:
登録されたメタ文字自体に「結合文字（絵文字の組み合わせや、濁点結合など）」が含まれている場合。
👉 GraphemeCursor を選択。

------------------------------
## 4. 責任分界点を極限まで細分化した「CharCursorSelector」の改修
このオンデマンド・ハイブリッド論理を反映し、原稿への不要なスキャンを完全に排除した、DRYかつ単一責任の CharCursorSelector の実装です。

```typescript
import { CharCursor, CodeUnitCursor, CodePointCursor, GraphemeCursor, TokenizerCursor } from './cursor.js';

/**
 * すべてのトークナイザ構成部品からメタ文字を一元的に収集・管理する責任
 */
export class MetaCharRegistry {
    private readonly chars: Set<string> = new Set();

    /** メタ文字をレジストリに登録する */
    register(metaChar: string): void {
        if (metaChar.length === 0) return;
        this.chars.add(metaChar);
    }

    /** 登録されたすべてのユニークなメタ文字のリストを返す */
    getRegisteredChars(): string[] {
        return Array.from(this.chars);
    }
}

/**
 * メタ文字の構成情報「だけ」から、原稿をスキャンすることなく最速のカーソルを論理的に決定する責任
 */
export class CharCursorSelector {
    private readonly segmenter: Intl.Segmenter;

    constructor() {
        this.segmenter = new Intl.Segmenter(undefined, { granularity: 'grapheme' });
    }

    /**
     * 登録されたメタ文字の複雑さに応じて、最適なカーソル戦略を選択する
     */
    select(text: string, registry: MetaCharRegistry): TokenizerCursor {
        const metaChars = registry.getRegisteredChars();
        
        let requireGrapheme = false;
        let requireCodePoint = false;

        for (const char of metaChars) {
            // メタ文字を人間の見た目の1字（書記素）ごとに分解して検証
            const segments = Array.from(this.segmenter.segment(char));
            
            for (const segment of segments) {
                const graphemeStr = segment.segment;
                
                // 論理1: 見た目の1字の中に、複数のコードポイント（結合文字・合字）が含まれているか
                // 根拠: Array.from("👨‍👩‍👧").length は 5 になるため、1つの書記素に複数字が含まれていると判定できる
                if (Array.from(graphemeStr).length > 1) {
                    requireGrapheme = true;
                    break;
                }
                
                // 論理2: 見た目の1字の内部CodeUnit長が 2（サロゲートペア）であるか
                if (graphemeStr.length > 1) {
                    requireCodePoint = true;
                }
            }
            if (requireGrapheme) break;
        }

        // 決定論理: メタ文字に存在しない複雑さのカーソルは、地の文にそれがあっても使う必要がない（隔離の法則）
        let selectedImpl: CharCursor;
        if (requireGrapheme) {
            selectedImpl = new GraphemeCursor(text);
        } else if (requireCodePoint) {
            selectedImpl = new CodePointCursor(text);
        } else {
            // すべてのメタ文字がASCIIなどの1コードユニットで完結している場合、最速の駆動を選択
            selectedImpl = new CodeUnitCursor(text);
        }

        return new TokenizerCursor(text, selectedImpl);
    }
}
```

これまでの議論、および単一責任の原則・DRY原則・動的ゲッターの型安全性を統合した、tokenizer/core/char-cursor-selector.ts のクリーンなTypeScriptコードです。

```typescript
import { CharCursor, CodeUnitCursor, CodePointCursor, GraphemeCursor, TokenizerCursor } from './cursor.js';
import { MetaCharRegistry } from './meta-char-registry.js';

/**
 * 収集されたメタ文字のUnicode特性を分析し、最適なカーソル戦略を決定・生成する責任クラス
 */
export class CharCursorSelector {
    private readonly segmenter: Intl.Segmenter;

    constructor() {
        // 国際化API（Intl.Segmenter）のインスタンス化をこのSelector内に閉じ込める（単一責任）
        this.segmenter = new Intl.Segmenter(undefined, { granularity: 'grapheme' });
    }

    /**
     * 登録されたメタ文字一覧から、論理的に最も効率的かつ安全なカーソルを選択して生成する
     */
    select(text: string, registry: MetaCharRegistry): TokenizerCursor {
        const metaChars = registry.getRegisteredChars();
        
        let requireGrapheme = false;
        let requireCodePoint = false;

        for (const char of metaChars) {
            // メタ文字を人間の見た目の1字（書記素クラスター）ごとに分解して検証
            const segments = Array.from(this.segmenter.segment(char));
            
            for (const segment of segments) {
                const graphemeStr = segment.segment;
                
                // 論理1: 見た目の1字の中に、複数のコードポイント（ZWJ合字、濁点結合など）が含まれているか
                // 根拠: Array.from("👨‍👩‍👧").length は 5 になるため、1つの書記素の中に複数のコードポイントがあると判定可能
                if (Array.from(graphemeStr).length > 1) {
                    requireGrapheme = true;
                    break;
                }
                
                // 論理2: 見た目の1字の内部CodeUnit長が 2（サロゲートペア）であるか
                if (graphemeStr.length > 1) {
                    requireCodePoint = true;
                }
            }
            // 最大単位がGrapheme（書記素）に達したら、これ以上のループ検証は不要なため即座に抜ける（最適化）
            if (requireGrapheme) {
                break;
            }
        }

        // 確定した大きい方の単位に合わせて、具象カーソルを割り当てる
        let selectedImpl: CharCursor;
        if (requireGrapheme) {
            selectedImpl = new GraphemeCursor(text);
        } else if (requireCodePoint) {
            selectedImpl = new CodePointCursor(text);
        } else {
            // すべてのメタ文字が1コードユニット（ASCIIなど）で完結している場合、最速のカーソルを採用
            selectedImpl = new CodeUnitCursor(text);
        }

        return new TokenizerCursor(text, selectedImpl);
    }
}

## 3. Factoryへの名称変更と引数の論理検証## Selector から Factory への変更の妥当性
CursorFactory への変更を支持します。
「Selector（選択者）」は既存の選択肢から選ぶだけというニュアンスが強いですが、今回のクラスは「引数（メタ文字情報など）に基づいて、内部で適切なインスタンスを組み立てて生成（製造）して返す」という明確な責任を持つため、Factory の名称が100%合致します。
## 引数の順序と「原稿が必須であること」の妥当性検証
提案された順序：CursorFactory.get(manuscript, separator, registry);
「原稿がないとカーソルが作れないが、これで良いか？」というご懸念について検証した結果、論理的に全く問題ありません。むしろこれが正しい設計です。

* 論理: カーソル（StringCursor）は、特定の文字列（原稿）の現在位置を指し示すという状態（State）を持つオブジェクトです。原稿が存在しない状態のカーソルというのは、中身の入っていない空のインデックスポインタに過ぎず、オブジェクトとして機能しません。したがって、インスタンス生成（get）の瞬間に manuscript が必須となるのは、オブジェクト指向の論理上、極めて自然で堅牢な制約です。

```typescript
export interface CharCursor {
    readonly isEof: boolean;
    peekChar(offset?: number): string;
    advance(charCount?: number): void;
    getAbsolutePosition(): number;
}

export class CodeUnitCursor implements CharCursor {
    private idx: number = 0;
    constructor(private readonly text: string) {}
    get isEof(): boolean { return this.idx >= this.text.length; }
    getAbsolutePosition(): number { return this.idx; }
    peekChar(offset: number = 0): string {
        if (this.isEof) return '';
        const target = this.idx + offset;
        if (target >= this.text.length) return '';
        return this.text.charAt(target);
    }
    advance(charCount: number = 1): void { this.idx += charCount; }
}

/**
 * サロゲートペア（16bit×2）の泣き別れを防ぐ責任に特化したカーソル
 */
export class SurrogatePairCursor implements CharCursor {
    private idx: number = 0;
    constructor(private readonly text: string) {}
    get isEof(): boolean { return this.idx >= this.text.length; }
    getAbsolutePosition(): number { return this.idx; }
    private calculateTargetPosition(charOffset: number): number {
        let tempIdx = this.idx;
        let count = 0;
        while (count < charOffset && tempIdx < this.text.length) {
            const code = this.text.codePointAt(tempIdx);
            tempIdx += (code !== undefined && code > 0xFFFF) ? 2 : 1;
            count++;
        }
        return tempIdx;
    }
    peekChar(offset: number = 0): string {
        if (this.isEof) return '';
        const targetIdx = this.calculateTargetPosition(offset);
        if (targetIdx >= this.text.length) return '';
        const code = this.text.codePointAt(targetIdx);
        return code !== undefined ? String.fromCodePoint(code) : '';
    }
    advance(charCount: number = 1): void { this.idx = this.calculateTargetPosition(charCount); }
}

export class GraphemeCursor implements CharCursor {
    private idx: number = 0;
    constructor(
        private readonly text: string,
        private readonly segmenter: Intl.Segmenter // Factoryから共通インスタンスをDI
    ) {}
    get isEof(): boolean { return this.idx >= this.text.length; }
    getAbsolutePosition(): number { return this.idx; }
    private calculateTargetPosition(graphemeOffset: number): number {
        if (graphemeOffset === 0) return this.idx;
        const segments = this.segmenter.segment(this.text.substring(this.idx));
        let count = 0;
        let relativeIndex = 0;
        for (const segment of segments) {
            if (count === graphemeOffset) break;
            relativeIndex += segment.segment.length;
            count++;
        }
        return this.idx + relativeIndex;
    }
    peekChar(offset: number = 0): string {
        if (this.isEof) return '';
        const start = this.calculateTargetPosition(offset);
        const end = this.calculateTargetPosition(offset + 1);
        return this.text.substring(start, end);
    }
    advance(charCount: number = 1): void { this.idx = this.calculateTargetPosition(charCount); }
}

export class TokenizerCursor implements CharCursor {
    constructor(
        private readonly text: string,
        private readonly cursorImpl: CharCursor
    ) {}
    get isEof(): boolean { return this.cursorImpl.isEof; }
    getAbsolutePosition(): number { return this.cursorImpl.getAbsolutePosition(); }
    peekChar(offset: number = 0): string { return this.cursorImpl.peekChar(offset); }
    advance(charCount: number = 1): void { this.cursorImpl.advance(charCount); }
    substring(start: number, end: number): string { return this.text.substring(start, end); }
}
```

## tokenizer/core/cursor-factory.ts
MetaCharRegistry から収集したメタ文字、および渡された可変の Separator の中身をDRYにスキャンし、Intl.Segmenter の管理まで一元化した製造責任クラスです。

```typescript
import { CharCursor, CodeUnitCursor, SurrogatePairCursor, GraphemeCursor, TokenizerCursor } from './cursor.js';
import { MetaCharRegistry } from './meta-char-registry.js';
import { Separator } from './separator.js';

export class CursorFactory {
    private readonly segmenter: Intl.Segmenter;

    /**
     * コンストラクタで上位（言語環境）からSegmenterを受け取れるように設計（DI）
     * 指定がない場合は環境デフォルトのgraphemeセグメンタを自給自足する
     */
    constructor(customSegmenter?: Intl.Segmenter) {
        this.segmenter = customSegmenter ?? new Intl.Segmenter(undefined, { granularity: 'grapheme' });
    }

    /**
     * 原稿、セパレータ、レジストリを監視し、最適なカーソルオブジェクトを製造（get）して返す責任
     */
    get(manuscript: string, separator: Separator, registry?: MetaCharRegistry): TokenizerCursor {
        // 1. レジストリからメタ文字一覧を取得（なければ空配列）
        const metaChars = registry ? registry.getRegisteredChars() : [];
        
        // 2. セパレータ自身のメタ文字（char）も検証対象に含めるため合流させる（DRY）
        if (separator.char) {
            metaChars.push(separator.char);
        }

        let requireGrapheme = false;
        let requireSurrogatePair = false;

        for (const char of metaChars) {
            const segments = Array.from(this.segmenter.segment(char));
            
            for (const segment of segments) {
                const graphemeStr = segment.segment;
                
                // 結合文字（合字）の検出
                if (Array.from(graphemeStr).length > 1) {
                    requireGrapheme = true;
                    break;
                }
                
                // サロゲートペアの検出
                if (graphemeStr.length > 1) {
                    requireSurrogatePair = true;
                }
            }
            if (requireGrapheme) break;
        }

        // 3. 複雑度に応じた具象カーソルの製造（製造責任の完結）
        let selectedImpl: CharCursor;
        if (requireGrapheme) {
            selectedImpl = new GraphemeCursor(manuscript, this.segmenter);
        } else if (requireSurrogatePair) {
            selectedImpl = new SurrogatePairCursor(manuscript);
        } else {
            selectedImpl = new CodeUnitCursor(manuscript);
        }

        return new TokenizerCursor(manuscript, selectedImpl);
    }
}
